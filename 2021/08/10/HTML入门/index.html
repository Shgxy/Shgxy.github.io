基本标签

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>基本标签学习</title>
</head>
<body>
<!--标题标签-->
<h1>一级标签</h1>
<h2>二级标签</h2>
<h3>三级标签</h3>
<h4>四级标签</h4>
<h5>五级标签</h5>
<h6>六级标签</h6>

<!--段落标签-->
<p>A B</p>
<p>C D</p>
<p>E F</p>

<!--换行标签-->
A B <br/>
C D <br/>

<!--水平线标签-->
<p>水平线:</p>
<hr/>

<!--字体样式标签-->
<h3>字体样式标签</h3>
粗体：<strong>啊啊啊啊</strong>
<br/>
斜体：<em>啊啊啊啊</em>
<br/>

<!--特殊符号-->
空&nbsp;&nbsp;&nbsp;&nbsp;格：
<br/>
&gt;
<br/>
&lt;
<br/>
&copy;版权所有shgxy

</body>
</html>

链接标签

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>链接标签学习</title>
</head>
<body>
<!--使用name作为标记-->
<a name="top">顶部</a>

<!--a标签
href：必填，表示要跳转到哪个页面
target：表示窗口在哪里打开
        _blank：在新标签中打开
        _self：在当前标签页中打开
-->
<a href="图像标签.html" target="_blank">点我获取图像</a>
<a href="https://shgxy.github.io" target="_self">点我跳转Blog</a>
<br>
<a href="https://shgxy.github.io"><img src="../Resources/image/test0810.png" alt="QQ截图" title="悬浮文字"width="264"height="145"></a>


<!--锚标签
1.需要一个标记
2.跳转到标记
#
-->
<a href="#top">点击回到顶部</a>

<!--功能性链接
邮件链接：mailto:
QQ链接：需要打开QQ推广
-->
<a href="mailto:123456@qq.com">邮件链接测试</a>
<a target="_blank" href="http://wpa.qq.com/msgrd?v=3&uin=&site=qq&menu=yes">

</body>

    <li>啊啊啊啊</li>
    <li>做着做着</li>
    <li>详细信息</li>

<ul>
    <li>去去去去</li>
    <li>溜了溜了</li>
    <li>啊啊啊啊</li>
    <li>做着做着</li>
    <li>详细信息</li>

dt：列表名称
dd：列表内容

-->
<dl>

    <dd>溜了溜了</dd>
    <dd>啊啊啊啊</dd>
    <dt>做着做着</dt>
    <dd>详细信息</dd>
    <dd>匆匆匆匆</dd>
</dl>

</body>
</html>

表格

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>表格学习</title>
</head>
<body>
<!--表格
行：tr
列：td
-->

<table border="1px">
    <tr>
        <!--colspan 跨列-->
        <td colspan="4" >1-1</td>
    </tr>
    <tr>
        <!--rowspan 跨行-->
        <td rowspan="3">2-1</td>
        <td>2-2</td>
        <td>2-3</td>
        <td>2-4</td>
    </tr>
    <tr>
        <td>3-1</td>
        <td>3-2</td>
        <td>3-3</td>
    </tr>
</table>

</body>
</html>

表单

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>登陆注册</title>
</head>
<body>
<!--表单 form
action：表单提交的位置，可以是网站，也可以是一个请求处理地址
method：post,get 提交方式
       get:可以在url中看到信息，高效，不安全
       post：可以传输大文件
-->
<form action="1.我的第一个HTML5网页.html" method="get">
    <!--文本输入框：<input type="text" name="username">
        value="shgxy" 默认初始值
        maxlength="8" 最长能写几个字符
        size="30"     文本框的长度
		readonly	  只读,即删不掉"admin"
		placeholder	  
    -->
    <p>用户：<input type="text" name="username" value="admin" placeholder="请输入用户名"></p>
    <p>密码：<input type="password" name="password"></p>

    <!--单选框标签
        input type="radio"
        radio放在同一组要求name的值一致，否则出现多选的情况
        value：单选框的值,是提交表单时传输的数据
        name：表示组
		disabled：表示不能选
    -->
    <p>性别：
        <input type="radio" value="boy" name="sex">郭楠
        <input type="radio" value="girl" name="sex" checked>幕刃
    </p>

    <!--多选框
        input type="checkbox"
    -->
    <p>爱好：
        <input type="checkbox" name="hobby" value="sing">唱
        <input type="checkbox" name="hobby" value="jump">跳
        <input type="checkbox" name="hobby" value="rap" checked>说唱
        <input type="checkbox" name="hobby" value="basketball">蝻球
    </p>

    <!--按钮
        input type="checkbox"   普通按钮
        input type="image"      图像按钮
        input type="submit"     提交按钮
        input type="reset"      重置按钮
    -->
    <p>按钮：
        <input type="button" name="btn" value="点击多一个老婆">
<!--        <input type="image" src="../Resources/image/test0810.png">-->
    </p>

    <!--下拉框，列表框
    -->
    <p>下拉框：
        <select name="列表名称">
            <option value="cn">CN</option>
            <option value="en">EN</option>
            <option value="cs" selected>CS</option>
        </select>
    </p>
    
    <!--文本域
        cols="30" rows="10"
    -->
    <p>投诉客服：
        <textarea name="textarea" cols="30" rows="10">文本域默认值</textarea>
    </p>

    <!--文件域
        input type="file"
    -->
    <p>
        <input type="file" name="getFile">
        <input type="button" name="upload" value="上传">
    </p>

    <!--邮件验证-->
    <p>邮箱：
        <input type="email" name="email">
    </p>

    <!--URL-->
    <p>URL：
        <input type="url" name="url">
    </p>

    <!--数字-->
    <p>Q币充值破解：
        <input type="number" name="num" max="100" min="0" step="0.1">
    </p>

    <!--滑块-->
    <p>破解速度：
        <input type="range" name="voice" min="0" max="100" value="61.8">
    </p>

    <!--搜索框-->
    <p>搜索：
        <input type="search" name="search">
    </p>
    
    <p>
        <input type="submit" value="登陆">
        <input type="reset" value="清空表单">
    </p>

</form>

</body>
</html>
